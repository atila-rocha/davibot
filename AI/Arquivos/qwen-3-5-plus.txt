## 🔍 Análise da Integração no Consumer

Você tem razão em destacar isso. O código que sugeri usa **regex dentro de uma única função**, o que **não segue o padrão de pattern matching em cláusulas de função** que você já tem no seu Consumer.

### ❌ Problema na Implementação Anterior

```elixir

defp handle_message(%{content: content} = msg) do
```

Isso viola o princípio de **despacho via pattern matching em cláusulas de função**.

### ✅ Jeito Correto (Alinhado com `!calabreso`)

O Consumer deve usar **múltiplas cláusulas de função** com pattern matching direto no `content`:

```elixir
# lib/davibot/consumer.exs
defmodule Davibot.Consumer do
  @moduledoc """
  handler de eventos - despacha os comandos
  Controller para os comandos - recebe as entradas e direciona para as funções commands
  """

  use Nostrum.Consumer
  require Logger

  alias Davibot.Commands

  def handle_event({:MESSAGE_CREATE, msg, _ws}) do
    handle_message(msg)
  end

  def handle_event(_event, state), do: {:noreply, state}

  # ✅ Pattern matching DIRETO no content - múltiplas cláusulas
  defp handle_message(%{content: "!calabreso"} = msg) do
    IO.inspect("CALABRESO MESSAGE ACIONADO")
    Commands.calabreso(msg)
  end

  defp handle_message(%{content: "!roleta-russa " &lt; num_str} = msg) do
    Commands.roleta_russa(msg.guild_id, msg.channel_id, String.trim(num_str), msg)
  end

  defp handle_message(%{content: "!roleta_russa " &lt; num_str} = msg) do
    Commands.roleta_russa(msg.guild_id, msg.channel_id, String.trim(num_str), msg)
  end

  defp handle_message(%{content: "!roleta russa " &lt; num_str} = msg) do
    Commands.roleta_russa(msg.guild_id, msg.channel_id, String.trim(num_str), msg)
  end

  # Catch-all para mensagens que não são comandos
  defp handle_message(_msg), do: :ok
end
```

### 📋 Vantagens Dessa Abordagem

1. **Despacho nativo do Elixir** - o pattern matching faz o dispatch, não um `case`
2. **Alinhado com `!calabreso`** - segue o mesmo padrão do seu código existente
3. **Mais performático** - pattern matching é otimizado pelo BEAM
4. **Mais legível** - cada comando é uma cláusula explícita
5. **Fácil de testar** - pode testar cada cláusula isoladamente

### ⚠️ Trade-off

**Limitação:** Pattern matching direto não suporta variações de caixa alta/baixa ou espaços extras. Se quiser suportar `!ROleta-RuSSa`, precisaria de:

```elixir
defp handle_message(%{content: content} = msg) when byte_size(content)  0 do
  content_lower = String.downcase(content)
  
  cond do
    content_lower == "!calabreso" -
      Commands.calabreso(msg)
    String.starts_with?(content_lower, "!roleta-russa ") -
      num_str = String.replace_leading(content_lower, "!roleta-russa ", "") | String.trim()
      Commands.roleta_russa(msg.guild_id, msg.channel_id, num_str, msg)
    true -
      :ok
  end
end
```

Mas isso **já não é pattern matching puro** — é um híbrido com `cond`.

### 🎯 Recomendação

Para manter a **pureza do pattern matching** e alinhamento com seu código atual:

1. Use **múltiplas cláusulas** para variações comuns (hífen, underscore, espaço)
2. Documente que o comando é **case-sensitive** (padrão do seu bot)
3. Se precisar de case-insensitive, migre **todos os comandos** para o approach com `cond`

```elixir
defmodule Davibot.Commands do
  require Logger
  alias Nostrum.Api
  alias Nostrum.Struct.Embed

  @doc """
  Comando !roleta-russa - bane N usuários aleatórios do servidor
  """
  def roleta_russa(msg) do
    with {:ok, num_str} &lt;- extract_num_str(msg.content),
         {n, _} &lt;- Integer.parse(num_str),
         true &lt;- n  0,
         {:ok, members} &lt;- Api.guild_members(msg.guild_id, %{limit: 1000}),
         {:ok, guild} &lt;- Api.get_guild(msg.guild_id),
         filtered &lt;- members
                     | Enum.filter(&(!&1.user.bot))
                     | Enum.filter(&(&1.user.id != guild.owner_id)),
         true &lt;- length(filtered) = n,
         banned &lt;- Enum.take_random(filtered, n),
         usernames &lt;- Enum.reduce(banned, [], &ban_and_log(&1, &2, msg.guild_id))
                     | Enum.reverse() do
      success_embed(usernames)
    else
      :invalid_num - error_embed("Número inválido! Use: `!roleta-russa N`")
      false - error_embed("Número deve ser inteiro positivo ou membros insuficientes")
      {:error, reason} - error_embed("Erro na API: #{inspect(reason)}")
    end
  end

  defp extract_num_str(content) do
    case Regex.run(~r/!(roleta[-_\s]+russa)\s+(\d+)/i, content) do
      [_, _, num_str] - {:ok, num_str}
      _ - :invalid_num
    end
  end

  defp ban_and_log(member, acc, guild_id) do
    case Api.ban_member(guild_id, member.user.id, %{delete_message_days: 0}) do
      {:ok} -
        Logger.info("Roleta-russa: Banned #{member.user.username} (#{member.user.id})")
        [member.user.username | acc]
      _ -
        acc
    end
  end

  defp success_embed(usernames) do
    %Embed{
      title: "🔫 Roleta Russa!",
      description: "💥 Banidos:\n" &lt; Enum.join(usernames, "\n"),
      color: 0xFF0000
    }
  end

  defp error_embed(reason) do
    %Embed{
      title: "❌ Erro",
      description: reason,
      color: 0xFF9900
    }
  end
end
```

```elixir
defmodule Davibot.Commands do
  alias Nostrum.Api
  alias Nostrum.Struct.Embed
  require Logger

  def roleta_russa(msg) do
    with {:ok, n} &lt;- parse_number(msg.content),
         {:ok, guild} &lt;- Api.get_guild(msg.guild_id),
         {:ok, members} &lt;- Api.get_guild_members(msg.guild_id, limit: 1000),
         filtered &lt;- filter_members(members, guild.owner_id),
         true &lt;- length(filtered) = n,
         selected &lt;- Enum.shuffle(filtered) | Enum.take(n),
         :ok &lt;- perform_bans(msg.guild_id, selected) do
      Logger.info("Roleta Russa: banned #{n} members in guild #{msg.guild_id}")
      success_embed(n)
    else
      {:error, reason} - error_embed(reason)
      false - error_embed("Not enough eligible members")
    end
  end

  defp parse_number(content) do
    content
    | String.trim()
    | Regex.run(~r/\d+/)
    | case do
      [num_str] -
        case Integer.parse(num_str) do
          {n, _} when n  0 and n &lt;= 500 - {:ok, n}
          _ - {:error, "Invalid positive integer (1-500)"}
        end
      _ - {:error, "No valid number found"}
    end
  end

  defp filter_members(members, owner_id) do
    Enum.filter(members, fn member -
      !member.user.bot and member.user.id != owner_id
    end)
  end

  defp perform_bans(guild_id, selected) do
    Enum.each(selected, &ban_member(guild_id, &1))
    :ok
  end

  defp ban_member(guild_id, member) do
    case Api.ban_user(guild_id, member.user.id, delete_message_days: 0) do
      {:ok, _} -
        Logger.info("Banned user #{member.user.id} (#{member.user.username})")
      {:error, reason} -
        Logger.warning("Failed to ban #{member.user.id}: #{inspect(reason)}")
    end
  end

  defp success_embed(n) do
    %Embed{
      title: "🎰 Roleta Russa!",
      description: "💥 Banidos #{n} membros aleatórios com sucesso!",
      color: 0xFF4444
    }
  end

  defp error_embed(reason) do
    %Embed{
      title: "❌ Erro na Roleta Russa",
      description: reason,
      color: 0xFF0000
    }
  end
end
```

```elixir
defmodule Davibot.Commands do
  alias Nostrum.Api
  alias Nostrum.Struct.Embed
  alias Nostrum.Struct.Guild
  require Logger

  @doc """
  Obtém o bot_id via Api.Self.application_information()
  Retorna {:ok, bot_id} ou {:error, reason}
  """
  def get_bot_id do
    with {:ok, %{bot: %{id: bot_id}}} &lt;- Api.Self.application_information() do
      {:ok, bot_id}
    else
      _ - {:error, "Falha ao obter ID do bot"}
    end
  end

  @doc """
  Extrai número do comando com regex genérico
  """
  def parse_number(content) do
    content
    | String.trim()
    | Regex.run(~r/!(roleta[-_\s]+russa)\s+(\d+)/i)
    | case do
      nil - {:error, "Formato inválido. Uso: !roleta-russa "}
      [_, _, num_str] -
        case Integer.parse(num_str) do
          {count, ""} when count  0 - {:ok, count}
          _ - {:error, "Número inválido"}
        end
    end
  end

  @doc """
  Filtra membros removendo dono e bot, retorna N aleatórios
  Usa member.user_id (não member.user.id)
  """
  def filter_members(guild_id, count, owner_id, bot_id) do
    guild_id
    | Api.guild_members(limit: 1000)
    | case do
      {:ok, members} -
        members
        | Enum.filter(&(&1.user_id != owner_id and &1.user_id != bot_id))
        | Enum.take_random(count)
      {:error, _reason} - []
    end
  end

  @doc """
  Executa bans em todos membros sorteados
  """
  def perform_bans(to_ban, guild_id) do
    to_ban
    | Enum.each(&ban_member(&1.user_id, guild_id, "Perdeu na roleta-russa! 💥"))
    :ok
  end

  defp ban_member(user_id, guild_id, reason) do
    Logger.info("Banning user #{user_id} from guild #{guild_id}: #{reason}")
    
    guild_id
    | Api.ban(user_id, 0, reason)
    | case do
      {:ok, _} - :ok
      {:error, err} - Logger.error("Failed to ban #{user_id}: #{inspect(err)}")
    end
  end

  defp success_embed(count) do
    %Embed{
      title: "💥 Roleta Russa Concluída!",
      description: "Baniu **#{count}** membro(s) aleatório(s)!",
      color: 0xFF0000
    }
  end

  defp error_embed(reason) do
    %Embed{
      title: "❌ Erro na Roleta",
      description: reason,
      color: 0xFF0000
    }
  end

  @doc """
  Função principal - orquestra todo o fluxo com with
  """
  def roleta_russa(%{content: content, guild_id: guild_id} = msg) when not is_nil(guild_id) do
    with {:ok, count} &lt;- parse_number(content),
         {:ok, bot_id} &lt;- get_bot_id(),
         {:ok, guild} &lt;- Api.get_guild(guild_id),
```

## 🔍 Explicação do `tap/2`

### 📋 O Que É

`tap/2` é uma função do Elixir que **executa um efeito colateral** (como loggar) e **retorna o valor original** inalterado.

### 🔧 Assinatura

```elixir
tap(value, function)
```

- `value` - o valor que passa pelo pipe
- `function` - função que recebe o valor, executa algo, e **não precisa retorná-lo**

### 💡 Por Que Usei

No comando `!sair`, preciso:
1. ✅ Chamar `Api.leave_guild(guild_id)`
2. ✅ Loggar a saída com `Logger.info`
3. ✅ Manter o fluxo do pipe sem quebrar

**Sem `tap`:**
```elixir
guild_id
| Api.leave_guild()
Logger.info("Davibot deixou o servidor #{guild_id}")  # ← Quebra o pipe!
```

**Com `tap`:**
```elixir
guild_id
| Api.leave_guild()
| tap(fn _ - Logger.info("Davibot deixou o servidor #{guild_id}") end)
# ← O valor original continua fluindo pelo pipe
```

### 🎯 Funcionamento Prático

```elixir
# Exemplo simples
5
| Integer.to_string()
| tap(fn s - IO.puts("Convertido: #{s}") end)
| String.upcase()
# Resultado: "FIVE"
# Efeito colateral: imprimiu "Convertido: 5" no console
```

### 📝 No Seu Código

```elixir
guild_id
| Api.leave_guild()
| tap(fn _ - Logger.info("Davibot saindo do servidor #{guild_id}: #{msg_text}") end)
```

**Fluxo:**
1. `guild_id` → entra no pipe
2. `Api.leave_guild(guild_id)` → bot sai do servidor
3. `tap(...)` → logga a mensagem **e retorna o resultado de `leave_guild`**
4. Próxima função no pipe recebe o resultado original (se houver)

### ✅ Vantagens

- **Mantém o pipe fluido** - não precisa quebrar para efeitos colaterais
- **Código mais legível** - efeito colateral fica explícito no fluxo
- **Idiomático Elixir** - `tap/2` foi feito exatamente para isso

### 🔄 Alternativa (menos elegante)

```elixir
result = Api.leave_guild(guild_id)
Logger.info("Davibot saiu: #{guild_id}")
result  # ← Precisa retornar manualmente
```

Com `tap`, isso fica tudo em uma cadeia de pipes limpa! 🎯

## 📌 Resumo: `tap/2`

### 🎯 O Que É

`tap/2` é uma função do Elixir que **executa um efeito colateral** (como loggar, imprimir, enviar notificação) e **retorna o valor original** inalterado.

### 🔧 Sintaxe

```elixir
valor | tap(fn x - faz_algo_com(x) end)
```

### 💡 Para Que Serve

**Problema que resolve:** Como fazer algo no meio de um pipe sem quebrar o fluxo?

**Sem `tap`:**
```elixir
guild_id
| Api.leave_guild()
IO.inspect("Saiu!")  # ← Quebra o pipe!
| proxima_funcao()  # ← Erro!
```

**Com `tap`:**
```elixir
guild_id
| Api.leave_guild()
| tap(fn _ - IO.inspect("Saiu!") end)  # ← Logga e continua
| proxima_funcao()  # ✅ Funciona!
```

### 📝 Exemplo Prático

```elixir
5
| Integer.to_string()        # "5"
| tap(fn s - IO.puts(s) end)  # Imprime "5" no console
| String.upcase()            # "5" (continua fluindo)
```

### ✅ Quando Usar

| Situação | Use `tap` |
|----------|-----------|
| Loggar/debuggar no meio do pipe | ✅ |
| Enviar notificação side-effect | ✅ |
| Validar sem alterar valor | ✅ |
| Transformar o valor | ❌ (use `map`/função normal) |

### 🎯 No Seu Código

```elixir

| Api.leave_guild()
| tap(fn _ - IO.inspect("Saiu: #{guild_id}", label: "SAÍDA") end)
```

**Resultado:**
- ✅ Bot sai do servidor
- ✅ Console imprime: `[SAÍDA] "Saiu: 123456789"`
- ✅ Pipe continua fluindo (se houver próxima função)

**Em uma frase:** `tap` é para **efeitos colaterais em cadeias de pipe** sem quebrar o fluxo! 🚀


## 🐛 Erro Identificado - Argumentos Duplicados no Pipe

O problema está no uso do **pipe operator** que está **duplicando o `channel_id`**.

### ❌ Código Problemático

```elixir

| Api.Message.create(channel_id, embeds: [confirm_embed])
```

**O que acontece:**
```elixir
# O pipe transforma em:
Api.Message.create(channel_id, channel_id, embeds: [confirm_embed])
#                                                  ↑
#                                          channel_id duplicado!
```

Isso chama `Api.Message.create/3` que **não existe**!

### ✅ Por Que `calabreso` Funciona

```elixir
# calabreso - CORRETO (sem pipe)
Api.Message.create(cid, embeds: [embed])
# Chama /2 ← existe!
```

```elixir
# sair - ERRADO (com pipe duplicando)
channel_id | Api.Message.create(channel_id, embeds: [embed])
# Chama /3 ← NÃO existe!
```

## 🔧 Solução - Não Use Pipe Aqui

O pipe não faz sentido quando você já vai passar os argumentos explicitamente.

## 📋 Regra do Pipe Operator

**Nunca use pipe quando:**
1. ❌ Você já vai passar todos os argumentos explicitamente
2. ❌ O primeiro argumento do pipe NÃO é o primeiro parâmetro da função

**Use pipe quando:**
1. ✅ O valor flui naturalmente como primeiro argumento
2. ✅ Você não precisa repetir o valor depois

### Exemplo Correto vs Incorreto

```elixir
# ❌ ERRADO - duplica argumento
channel_id | Api.Message.create(channel_id, embeds: [embed])

# ✅ CORRETO - sem pipe
Api.Message.create(channel_id, embeds: [embed])

# ✅ CORRETO - pipe faz sentido (valor flui como 1º argumento)
content
| String.trim()
| String.slice(6..-1)
```

```elixir
defmodule Davibot.Commands do
  alias Nostrum.Api
  alias Nostrum.Struct.Embed
  alias Nostrum.Struct.User
  require Logger

  # ... funções existentes (roleta_russa, sair, calabreso, etc.)

  @doc """
  Comando !avatar - retorna avatar do usuário com tamanho específico
  Uso: !avatar &lt;@usuario 
  """
  def avatar(%{content: content, channel_id: channel_id} = msg) do
    content
    | String.trim()
    | extract_avatar_args()
    | case do
      {:ok, user_id, size} -
        user_id
        | get_user_info()
        | build_avatar_url(size)
        | send_avatar_embed(channel_id, user_id)

      :error -
        error_embed = %Embed{
          title: "❌ Erro",
          description: "Uso: `!avatar &lt;@usuario ` (16-4096 px)",
          color: 0xFF0000
        }
        Api.create_message(channel_id, embeds: [error_embed])
    end
  end

  @doc """
  Extrai user_id da menção e tamanho do argumento
  Regex: &lt;@123456789 ou &lt;@!123456789
  """
  defp extract_avatar_args(content) do
    # Remove o comando "!avatar" e pega o resto
    args = content | String.replace_prefix("!avatar", "") | String.trim()

    # Regex para menção Discord: &lt;@123456789 ou &lt;@!123456789
    mention_regex = ~r/&lt;@!?(\d+)/
    size_regex = ~r/\d+/

    with [_, user_id_str] &lt;- Regex.run(mention_regex, args),
         [size_str] &lt;- Regex.run(size_regex, args),
         {user_id, _} &lt;- Integer.parse(user_id_str),
         {size, _} &lt;- Integer.parse(size_str),
         true &lt;- size = 16 and size &lt;= 4096 do
      {:ok, user_id, size}
    else
      _ - :error
    end
  end

  @doc """
  Obtém info do usuário via API ou Cache
  """
  defp get_user_info(user_id) do
    # Tenta pegar do cache primeiro, depois da API
    case Api.get_user(user_id) do
      {:ok, user} - {:ok, user}
      {:error, _} - :error
    end
  end

  @doc """
  Constrói URL do avatar (personalizado ou padrão)
  """
  defp build_avatar_url({:ok, user}, size) do
    user_id_str = to_string(user.id)

    avatar_url =
      cond do
        # Usuário tem avatar personalizado
        user.avatar != nil and user.avatar != "" -
          "https://cdn.discordapp.com/avatars/#{user_id_str}/#{user.avatar}.png?size=#{size}"

        # Usa avatar padrão do Discord (baseado no discriminator)
        true -
          disc_num = String.to_integer(user.discriminator) | rem(5)
          "https://cdn.discordapp.com/embed/avatars/#{disc_num}.png"
      end

    {:ok, avatar_url, user}
  end

  defp build_avatar_url(:error, _size), do: :error

  @doc """
  Envia embed com avatar
  """
  defp send_avatar_embed({:ok, avatar_url, user}, channel_id, _user_id) do
    username = user.username || user.global_name || "Usuário"

    embed = %Embed{
      title: "🖼️ Avatar de #{username}",
      image: %Embed.Image{url: avatar_url, width: size, height: size},
      color: 0x7289DA,
      footer: %Embed.Footer{text: "Tamanho: #{size}x#{size} px"}
    }

    Api.create_message(channel_id, embeds: [embed])
  end

  defp send_avatar_embed(:error, channel_id, _user_id) do
    error_embed = %Embed{
      title: "❌ Erro",
      description: "Não foi possível obter o avatar do usuário.",
      color: 0xFF0000
    }

    Api.create_message(channel_id, embeds: [error_embed])
  end

  # ... resto dos módulos existentes
```


- `(\d+)` - captura os dígitos do ID
- `` - fecha com ``

**Funciona para:**
- `&lt;@123456789` - menção normal
- `&lt;@!123456789` - menção com nickname
